#### // FIRST TABLE

CREATE TABLE STUDENT(

SID INT UNSIGNED NOT NULL,

SNMAE VARCHAR(20) NOT NULL,

PHONE BIGINT UNSIGNED NOT NULL,

CONSTRAINT U\_PH  UNIQUE(PHONE),

CONSTRAINT P\_SID PRIMARY KEY(SID),

CONSTRAINT C\_PH CHECK(LENGTH(PHONE)=10)

);



#### //BRANCH

&#x20;   ADDRESS VARCHAR(20) NOT NULL,

&#x20;   PINCODE INT UNSIGNED NOT NULL,

&#x20;   CONSTRAINT P\_BID PRIMARY KEY(BID),

&#x20;   CONSTRAINT U\_PIN UNIQUE(PINCODE),

&#x20;   CONSTRAINT C\_PIN CHECK(LENGTH(PINCODE)=6));



### //PRODUCT

CREATE TABLE PRODUCT(

PID INT NOT NULL,

PNAME VARCHAR(20) NOT NULL,

PRICE DECIMAL(5,2) NOT NULL,

QUANTITY INT NULL,

CONSTRAINT P\_PID PRIMARY KEY(PID),

CONSTRAINT C\_PRICE CHECK(PRICE>0),

CONSTRAINT C\_QUALITY CHECK(QUANTITY>0));

## //CUSTOMER

CREATE TABLE CUSTOMER(

CID INT UNSIGNED NOT NULL PRIMARY KEY AUTO\_INCREMENT,

CNAME VARCHAR(30) NOT NULL,

ADDRESS VARCHAR(100) NOT NULL,

PHONE BIGINT UNSIGNED NOT NULL UNIQUE CHECK(LENGTH(PHONE)=10),

EMAIL VARCHAR(40) NULL,

BALANCE DECIMAL(5,3) UNSIGNED NOT NULL DEFAULT "10",

GENDER ENUM("MALE","FEMALE") NOT NULL,

SID INT UNSIGNED NULL,

CONSTRAINT FOREIGN KEY(SID) REFERENCES STUDENT(SID)

);

DROP AUTO\_CREMENT THEN REMOVE PRIMARY KEY;

WE CANT DROP PRIMARY KEY FROM A TABLE IF PRIMARY KEY CLOUMN IS FOREIGN KEY IN ANOTHER TABLE.

DROP FORIGIN KEY AND THEN REMOVE PRIMARY KEY
