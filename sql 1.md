## //PRODUCT

CREATE TABLE PRODUCT(

PID INT NOT NULL,

PNAME VARCHAR(20) NOT NULL,

PRICE DECIMAL(5,2) NOT NULL,

QUANTITY INT NULL,

CONSTRAINT P\_PID PRIMARY KEY(PID),

CONSTRAINT C\_PRICE CHECK(PRICE>0),

CONSTRAINT C\_QUALITY CHECK(QUANTITY>0));

## // CUSTOMER

CREATE TABLE CUSTOMER(

CID INT UNSIGNED NOT NULL PRIMARY KEY AUTO\_INCREMENT,

CNAME VARCHAR(30) NOT NULL,

ADDRESS VARCHAR(100) NOT NULL,

PHONE BIGINT UNSIGNED NOT NULL UNIQUE CHECK(LENGTH(PHONE)=10),

EMAIL VARCHAR(40) NULL,

BALANCE DECIMAL(5,3) UNSIGNED NOT NULL DEFAULT "10",

GENDER ENUM("MALE","FEMALE") NOT NULL,

SID INT UNSIGNED NULL,

CONSTRAINT FOREIGN KEY(SID) REFERENCES STUDENT(SID)

);

## //PACKAGE

CREATE TABLE PACKAGES(

PACKAGE\_ID INT NOT NULL PRIMARY KEY,

DESTINATION\_ID INT,

PACKAGE\_NAME VARCHAR(100),

DURATION\_DAYS INT,

PRICE DECIMAL(10,2),

CONSTRAINT FOREIGN KEY(DESTINATION\_ID) REFERENCES DISTINATION(DESTINATION\_ID));



## //BOOKING



CREATE TABLE BOOKING(

BOOKING\_ID INT NOT NULL PRIMARY KEY,

CUSTOMER\_ID INT NULL,

PACKAGE\_ID INT NULL,

BOOKING\_DATE DATE NULL,

STATUS ENUM('CONFIRMED','CANCELLED','PENDING'),

CONSTRAINT FOREIGN KEY(CUSTOMER\_ID) REFERENCES CUSTOMER(CUSTOMER\_ID),

CONSTRAINT FOREIGN KEY(PACKAGE\_ID) REFERENCES PACKAGES(PACKAGE\_ID)

);
